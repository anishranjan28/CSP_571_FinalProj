2a. Re-label clusters in a stable order

To keep names stable across re-runs we reorder by median log_Monetary so cluster A is always the highers-spending group, C the lowest.

order_by_value <- cust |>
  group_by(cluster) |>
  summarise(med_logM = median(log_Monetary), .groups = "drop") |>
  arrange(desc(med_logM)) |>
  mutate(label = LETTERS[seq_len(n())]) |>
  select(cluster, label)

cust <- cust |>
  left_join(order_by_value, by = "cluster") |>
  mutate(cluster = factor(label, levels = LETTERS[seq_along(unique(label))])) |>
  select(-label)

9. A simple, interpretable rule for cluster membership

tree_fit <- rpart(
  cluster ~ log_Recency + log_Frequency + log_Monetary +
            log_UniqueProducts + log_AvgOrderValue,
  data = cust,
  method = "class",
  control = rpart.control(maxdepth = 4, cp = 0.005)
)

rpart.plot(tree_fit, type = 2, extra = 104, fallen.leaves = TRUE,
           main = "Decision rule approximating cluster membership")

pred <- predict(tree_fit, type = "class")
tab <- table(actual = cust$cluster, predicted = pred)
acc <- sum(diag(tab)) / sum(tab)

list(
  confusion_matrix = tab,
  training_accuracy = percent(acc, accuracy = 0.1)
)
